BINARY-SEARCH(L, low, high)
	i <- 0					//i is first index
	b <- length[L] - 1			//b is last index
	found <- FALSE				//found is final parameter

	WHILE i <= b and found = FALSE		
		m <- floor[(i+b)/2]		//calculate midpoint
		IF low <= L[m] <= high		//check m in range
			found <- TRUE		//break loop
	ELSE
		IF low < L[m]			//split list down
			b <- m - 1
		ELSE
			i <- m + 1		//split list upwards
	RETURN found				//return result
